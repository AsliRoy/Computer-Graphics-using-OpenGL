void myInit(void)
{
	glClearColor(0.0, 0.0, 0.0, 0.0);
	gluOrtho2D(0.0, 500.0, 0.0, 500.0);
}

void draw_pixel(int x, int y) {
	glBegin(GL_POINTS);
	glVertex2i(x, y);
	glEnd();
}

void drawP()
{
	glPointSize(1.0);
	glColor3f(1,1,1);
	glBegin(GL_LINES);
	glVertex2i(x1,y1);
	glVertex2i(x2,y2);	
	glEnd();

	glBegin(GL_LINES);
	glVertex2i(x2,y2);	
	glVertex2i(x3,y3);
	glEnd();

	
}
void draw_curve() {	
	//drawP();
	double u,x,y,f1,f2,f3;
	draw_pixel(x1, y1);
	draw_pixel(x3, y3);
	for(u=0.01;u<=1;u+=0.01)
	{
		f1 = 2*u*u - 3*u + 1;
		f2 = -4*u*u + 4*u ;
		f3 = 2*u*u - u;
		x = f1*x1 + f2*x2 + f3*x3;
		y = f1*y1 + f2*y2 + f3*y3;
		draw_pixel(x, y);
	}

	/*

void myDisplay() {
	draw_curve();
	glFlush();
}

void main(int argc, char **argv) {

	printf("Enter (x1, y1, x2, y2,x3,y3)\n");
	scanf("%d %d %d %d %d %d", &x1, &y1, &x2, &y2,&x3,&y3);

	glutInit(&argc, argv);
	glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
	glutInitWindowSize(500, 500);
	glutInitWindowPosition(0, 0);
	glutCreateWindow("Plane Curve");
	myInit();
	glutDisplayFunc(myDisplay);
	glutMainLoop();
}
